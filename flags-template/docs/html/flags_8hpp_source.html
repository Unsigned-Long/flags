467  OptionParser &setupParser(int argc, char const *argv[]) {
468  // first, generate the corresponding help document and version number according to the set options
469  if (this->_options.at(this->VERSION_OPTION_NAME)._arg->getArgv<StringArg>().empty()) {
470  this->autoGenerateVersion();
471  }
472  if (this->_options.at(this->HELP_OPTION_NAME)._arg->getArgv<StringArg>().empty()) {
473  this->autoGenerateHelp(argv[0]);
474  }
475 
476  // find all valid options in the parameter list and record their names and locations
477  std::vector<std::pair<std::string, std::size_t>> optionInfo;
478  std::unordered_set<std::string> optionsPassed;
479  for (int i = 1; i != argc; ++i) {
480  if (this->optionOrArgv(argv[i])) {
481  // is an option
482  auto optionName = std::string(argv[i]).substr(2);
483 
484  // is help or version options
485  if (optionName == this->HELP_OPTION_NAME) {
486  throw std::runtime_error(this->getOptionArgv<StringArg>(this->HELP_OPTION_NAME));
487  } else if (optionName == this->VERSION_OPTION_NAME) {
488  throw std::runtime_error(argv[0] + std::string(" version: ") +
489  this->getOptionArgv<StringArg>(this->VERSION_OPTION_NAME));
490  }
491 
492  optionInfo.push_back({optionName, i});
493  optionsPassed.insert(optionName);
494  }
495  }
496 
497  // check whether any missing options have not been passed in according to the properties of the set options
498  if (argc > 1) {
499  if (optionInfo.empty() || optionInfo.front().second > 1) {
500  optionsPassed.insert(this->DEFAULT_OPTION_NAME);
501  }
502  }
503  for (const auto &[name, option] : this->_options) {
504  if (option._prop == OptionProp::REQUIRED) {
505  auto iter = optionsPassed.find(name);
506  if (iter == optionsPassed.cend()) {
507  THROW_EXCEPTION(setupParser, "the option named '--" + name + "' is 'OptionProp::REQUIRED', but you didn't use it");
508  }
509  }
510  }
511 
512  // divide the parameter list according to the position of the option
513  // and set it to the value of the corresponding option
514  if (optionInfo.empty()) {
515  return *this;
516  }
517 
518  std::vector<std::string> strVec;
519  for (int i = 1; i != optionInfo.front().second; ++i) {
520  strVec.push_back(argv[i]);
521  }
522  if (auto iter = this->_options.find(this->DEFAULT_OPTION_NAME); iter != this->_options.end()) {
523  this->_options.at(this->DEFAULT_OPTION_NAME)._arg->setArgv(strVec);
524  }
525 
526  for (int i = 0; i < optionInfo.size() - 1; ++i) {
527  strVec.clear();
528  auto curOption = optionInfo.at(i);
529  auto nextOption = optionInfo.at(i + 1);
530  for (int j = curOption.second + 1; j < nextOption.second; ++j) {
531  strVec.push_back(argv[j]);
532  }
533  this->_options.at(curOption.first)._arg->setArgv(strVec);
534  }
535 
536  strVec.clear();
537  for (int j = optionInfo.back().second + 1; j < argc; ++j) {
538  strVec.push_back(argv[j]);
539  }
540  this->_options.at(optionInfo.back().first)._arg->setArgv(strVec);
541 
542  return *this;
543  }
